\begin{luacode}
 documentdata = documentdata or { }

 local stringformat = string.format
 local texsprint = tex.sprint
 local slot_of_name = luaotfload.aux.slot_of_name

 documentdata.fontchar = function (chr)
 local chr = slot_of_name(font.current(), chr, false)
 if chr and type(chr) == "number" then
 texsprint
 (stringformat ([[\char"%X]], chr))
 end
 end
\end{luacode}

\DeclareDocumentCommand{\identificationKey}{m m m m}{
	\newQuestion%
	\def\thisKey{#1}%
	\ifx\lastKey\thisKey\else\reversemarginpar\marginpar{\vskip -4mm\includegraphics[width=25mm]{keys/#1.pdf}}\fi%
	\ifnum#2=1 
		В знаке есть \expandafter\csname keyNomPl#1\endcsname?
	\else
		\newcount\adjustvalue
		\adjustvalue = #2
		\advance\adjustvalue by -1%
		Количество \expandafter\csname keyGenPl#1\endcsname\  в знаке больше \the\adjustvalue?
	\fi
	\par Да — \ref{#3}. Нет — \ref{#4}.%
	\global\def\lastKey{#1}%
}
