\begin{luacode}
 documentdata = documentdata or { }

 local stringformat = string.format
 local texsprint = tex.sprint
 local slot_of_name = luaotfload.aux.slot_of_name

 documentdata.fontchar = function (chr)
 local chr = slot_of_name(font.current(), chr, false)
 if chr and type(chr) == "number" then
 texsprint
 (stringformat ([[\char"%X]], chr))
 end
 end
\end{luacode}

\def\fontchar#1{\directlua{documentdata.fontchar "#1"}}

\DeclareDocumentCommand{\keyNames}{m m m}{
	\global\expandafter\edef\csname keyNomPl#1\endcsname{{#2}}
	\global\expandafter\edef\csname keyGenPl#1\endcsname{{#3}}
}

% O — круг (или овал, важно, чтобы полный, без разрывов)
% R — квадрат (или другой четырехугольник)
% C — дуга
% J — крючок (изогнутая в одну сторону линия со свободным концом)
% B — петля (любой формы)
% S — s-образный элемент (в любую сторону)
% G — спираль (в любую сторону)
% L — кочерга (любая линия с одним изломом и свободным концом)
% I — палка (прямая линия без изгиба)
% Q — маленькая черточка (короткая, неопределенной формы)
% Z — зигзаг или волнистая линия
% D — «ручка» (круглая или треугольная)
% A — треугольник
% K — точка, короткая черточка, маленький кружок, маленький наплыв

\DeclareDocumentCommand{\identificationKey}{m m m m}{
\section{}
\ifnum#2=1 
В знаке есть \expandafter\csname keyNomPl#1\endcsname?
\else
Количество \expandafter\csname keyGenPl#1\endcsname\  в знаке больше или равно #2?\\
\fi
Да — \ref{#3}. Нет — \ref{#4}.
}
